  rib &
  rib::operator=(
    rib const &rib_in)
  {
    if (this == &rib_in)
    {
      return *this;
    }
    else
    {
      this->disk::operator=(rib_in);
      this->m_width       = rib_in.m_width;
      this->m_angle       = rib_in.m_angle;
      return *this;
    }
  }

  bool
  rib::isApprox(
    rib const &rib_in,
    real       tolerance)
    const
  {

    return this->disk::isApprox(rib_in) &&
           acme::isApprox(this->m_width, rib_in.m_width, tolerance) &&
           acme::isApprox(this->m_angle, rib_in.m_angle, tolerance);
  }

  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

  bool
  rib::envelopRadial(
    segment const &segment_in,
    real          &radial_area,
    vec3          &radial_normal,
    real           tolerance)
    const
  {
    if (segment_in.isDegenerated())
    {
      radial_area   = 0.0;
      radial_normal = ZEROS_VEC3;
      return false;
    }
    else
    {
      real  rib_radius = this->radius();
      point rib_center(this->center());

      segment segment_tmp(rib_center + (segment_in.vertex(0) - rib_center).normalized() * rib_radius,  // Point A
                          rib_center + (segment_in.vertex(1) - rib_center).normalized() * rib_radius); // Point B

      real area_triangle = 0.5 * ((segment_in.vertex(1) - segment_in.vertex(0)).cross(rib_center - segment_in.vertex(0))).norm();
      // real area_sector   = rib_radius * rib_radius * std::acos(std::sqrt(4 * rib_radius * rib_radius - segment_tmp.length() * segment_tmp.length()) / (2.0 * rib_radius));
      real area_sector = rib_radius * rib_radius * std::asin(std::sqrt(segment_tmp.length() / (2.0 * rib_radius)));

      radial_area   = area_sector - area_triangle;
      radial_normal = segment_in.centroid() - (segment_tmp.centroid() - rib_center).normalized() * rib_radius;

      ENVE_ASSERT(area_triangle >= 0.0, "enve::rib::envelopRadial(segment, area, normal, tolerance): negative triangle area.\n");
      ENVE_ASSERT(area_sector >= 0.0, "enve::rib::envelopRadial(segment, area, normal, tolerance): negative rib sector area.\n");
      ENVE_ASSERT(radial_area >= 0.0, "enve::rib::envelopRadial(segment, area, normal, tolerance): negative radial area.\n");

      if (radial_area < tolerance)
      {
        radial_area   = 0.0;
        radial_normal = ZEROS_VEC3;
        return false;
      }
      else
      {
        return true;
      }
    }
  }
