#pragma once

#ifndef INCLUDE_ENVE_MESH
#define INCLUDE_ENVE_MESH

// Print progress to console while loading (large models)
#define ENVE_CONSOLE_OUTPUT

#include <acme.hh>
#include <acme_AABBtree.hh>
#include <acme_aabb.hh>
#include <acme_utils.hh>
#include <acme_triangle.hh>

#include "enve.hh"
#include "enve_triangleground.hh"

using namespace acme;

namespace enve
{

  namespace ground
  {

    /*\
     |                       _
     |   _ __ ___   ___  ___| |__
     |  | '_ ` _ \ / _ \/ __| '_ \
     |  | | | | | |  __/\__ \ | | |
     |  |_| |_| |_|\___||___/_| |_|
     |
    \*/

    class mesh
    {
    public:
      typedef std::shared_ptr<mesh const> ptr;
      typedef std::vector<mesh::ptr>      vecptr;

    private:
      triangleground::vecptr m_triangles;
      aabb::vecptr           m_bboxes;
      AABBtree::ptr          m_AABBtree;

    public:
      ~mesh(void);

      mesh(mesh const &) = delete;

      mesh(mesh const &&) = delete;

      mesh(void);

      mesh(
        triangleground::vecptr const &triangles
      );

      mesh(
        std::string const &path
      );

      mesh(
        std::string const &path,
        real               friction
      );

      void
      copy(
        mesh const &mesh_in
      );

      void
      clear(void);

      triangleground::vecptr const &
      vecptrTriangleground(void) const;

      triangleground::ptr
      ptrTriangleground(
        size_t i
      ) const;

      AABBtree::ptr const
      ptrAABBtree(void) const;

      void
      print(
        std::string const &path
      ) const;

      aabb::vecptr const &
      bboxes(void) const;

      size_t
      size(void) const;

      bool
      load(
        std::string const &path
      );

      bool
      load(
        std::string const &path,
        real               friction
      );

      bool
      intersection(
        AABBtree::ptr const     AABBtree,
        triangleground::vecptr &triangles
      ) const;

      bool
      intersection(
        aabb::vecptr const     &boxes,
        triangleground::vecptr &triangles
      ) const;

      bool
      intersection(
        aabb::ptr const         box,
        triangleground::vecptr &triangles
      ) const;

    private:
      void
      updateBBoxes(void);

      void
      generateVertices(
        std::vector<point> const &inodes,
        std::string const        &icurline,
        point                     overts[3]
      ) const;

      void
      split(
        std::string const        &in,
        std::vector<std::string> &out,
        std::string const        &token
      ) const;

      std::string
      tail(
        std::string const &in
      ) const;

      std::string
      firstToken(
        std::string const &in
      ) const;

      template <typename T>
      T const &
      element(
        std::vector<T> const &elements,
        std::string const    &index)
        const
      {
        integer id = std::stoi(index);
        ENVE_ASSERT(id > 0, "enve::element(): element indexes cannot be negative\n");
        return elements[id - 1];
      }
    };

  } // namespace ground

} // namespace enve

#endif
